\begin{frame}{Safety \& Compliance}
  \footnotesize

  \begin{columns}[T,onlytextwidth]
    \column{0.49\textwidth}
    \textbf{Main hazards}
    \begin{itemize}\setlength\itemsep{2pt}
      \item Hot nozzle / bed and moving printer axes
      \item Sharp edges after support removal or trimming
      \item Pinch points at the lead screw and moving carriage
      \item Sudden spring-back or cracking of polymer samples
      \item Sensor cables snagging on the rig
    \end{itemize}

    \column{0.49\textwidth}
    \textbf{Controls}
    \begin{itemize}\setlength\itemsep{2pt}
      \item Ventilated printing; allow printer to cool before handling
      \item Deburr samples; keep hands clear of compression path
      \item Wear safety glasses during every loading test
      \item One operator applies load slowly and stops at slip,
            crack, or permanent set
      \item Secure specimen / sensors and route cables clear
    \end{itemize}
  \end{columns}

  {\scriptsize
    \textbf{Compliance:} Benchtop mechanical experiment only; no
    chemical, radiation, or biological hazards.
    \textbf{Waste:} Failed prints and broken polymer samples are
    disposed as solid plastic waste.
    \textbf{Emergency:} Lab supervisor, makerspace / lab technician,
    campus security (ext.\ 8888).}
\end{frame}
