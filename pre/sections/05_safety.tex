\begin{frame}{Safety \& Compliance}
  \small
  \begin{itemize}\setlength\itemsep{2pt}
    \item \textbf{Hazards identified:}
      \begin{itemize}
        \item 3D printer: hot nozzle ($\sim 200\,{}^{\circ}\mathrm{C}$),
              moving parts, minor fume emission from PLA.
        \item Compression testing: spring-back of samples, pinch risk
              from lead screw mechanism.
      \end{itemize}
    \item \textbf{Mitigation measures:}
      \begin{itemize}
        \item Operate printer in ventilated area; do not touch nozzle
              during operation.
        \item Wear safety glasses during compression tests.
        \item Use hand wheel with controlled speed; never exceed
              elastic limit of samples.
      \end{itemize}
    \item \textbf{Regulatory status:}
          No chemicals, radiation, or biological agents involved.
          Not regulated under Malaysian Occupational Safety and
          Health Act 1994.
    \item \textbf{Waste disposal:}
          PLA is biodegradable; failed prints collected in general
          recycling bin.
    \item \textbf{Emergency contacts:}
          Lab supervisor; campus security (ext.\ 8888).
  \end{itemize}
\end{frame}
