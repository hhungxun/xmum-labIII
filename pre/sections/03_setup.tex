\begin{frame}{Theory Framework}
  
      The theoretical basis is divided into three parts:
      \begin{enumerate}
        \item \textbf{Topology optimization model}: convert structure design
            into a material distribution problem.
        \item \textbf{Mechanical constraints}: ensure the design is safe under
            static and cyclic loading.
      \item \textbf{Energy-based objective}: optimise stored elastic energy
            and energy release efficiency.
      \end{enumerate}


\end{frame}

%------------------------------------------------
\begin{frame}{Topology Optimization Model}
  \small
  \begin{columns}[T]
    \column{0.58\textwidth}
    \begin{itemize}\setlength\itemsep{3pt}
      \item The design domain is divided into finite elements.
      
      

      \item Each element has a density variable:
        \[
          \rho_e \in [0,1]
        \]
      \item $\rho_e=1$ represents solid material, while $\rho_e=0$
            represents void.
      \item The SIMP method connects density and material stiffness
            \cite{bendsoe2003topology, sigmund2001}:
        \[
          E_e = E_{\min}+\rho_e^{\,p}(E_0-E_{\min})
        \]
      \item The structural response is obtained from:
        \[
          \mathbf{K}(\boldsymbol{\rho})\mathbf{U}=\mathbf{F}
        \]
    \end{itemize}

    \centering

\begin{frame}{Mechanical and Physical Constraints}
  \small
  \begin{columns}[T]
    \column{0.58\textwidth}
    \begin{itemize}\setlength\itemsep{3pt}
      \item The optimised structure must be mechanically feasible,
            not only numerically optimal.
      \item \textbf{Static strength}: avoid yielding or fracture:
        \[
          \sigma_{\mathrm{vm}} \le \sigma_y
        \]
      \item \textbf{Fatigue constraint}: evaluate cyclic loading using
            a Goodman-type criterion \cite{zhao2023topology}:
        \[
          \frac{\sigma_a}{\sigma_e}
          +
          \frac{\sigma_m}{\sigma_{ut}}
          \le 1
        \]
      \item Mean and alternating stresses are:
        \[
          \sigma_a=\frac{\sigma_{\max}-\sigma_{\min}}{2},
          \qquad
          \sigma_m=\frac{\sigma_{\max}+\sigma_{\min}}{2}
        \]
      \item \textbf{Manufacturing}: density filtering controls minimum
            feature size and improves 3D printability.
    \end{itemize}

    \column{0.40\textwidth}
    \centering

\begin{frame}{Apparatus \& Test Conditions}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.49\textwidth}
    \textbf{Rig \& sensing}
    \begin{itemize}\setlength\itemsep{3pt}

      \item Linear guide rail and pushing plate
      \item Force sensor for applied load $F(t)$
      \item Motion sensor for displacement $x(t)$
      \item One-end clamped boundary condition
    \end{itemize}

    \column{0.49\textwidth}
    \textbf{Specimens \& analysis}
    \begin{itemize}\setlength\itemsep{3pt}
      \item FDM-printed PETG planar test structures
      \item Fixed footprint and equal-thickness specimens
      \item GNU Octave, FreeCAD, and Python for design / analysis
      \item Room temperature ($\sim\!25\,{}^{\circ}\mathrm{C}$)
      \item Quasi-static manual loading for force--displacement evaluation
    \end{itemize}
  \end{columns}
\end{frame}
