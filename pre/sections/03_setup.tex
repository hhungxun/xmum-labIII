\begin{frame}{Theory Framework}
  
      The theoretical basis is divided into three parts:
      \begin{enumerate}
        \item \textbf{Topology optimization model}: convert structure design
            into a material distribution problem.
        \item \textbf{Mechanical constraints}: ensure the design is safe under
            static and cyclic loading.
      \item \textbf{Energy-based objective}: optimise stored elastic energy
            and energy release efficiency.
      \end{enumerate}


\end{frame}

%------------------------------------------------
\begin{frame}{Topology Optimization Model}
  \small
  \begin{columns}[T]
    \column{0.58\textwidth}
    \begin{itemize}\setlength\itemsep{3pt}
      \item The design domain is divided into finite elements.
      
      

      \item Each element has a density variable:
        \[
          \rho_e \in [0,1]
        \]
      \item $\rho_e=1$ represents solid material, while $\rho_e=0$
            represents void.
      \item The SIMP method connects density and material stiffness
            \cite{bendsoe2003topology, sigmund2001}:
        \[
          E_e = E_{\min}+\rho_e^{\,p}(E_0-E_{\min})
        \]
      \item The structural response is obtained from:
        \[
          \mathbf{K}(\boldsymbol{\rho})\mathbf{U}=\mathbf{F}
        \]
    \end{itemize}

\begin{frame}{Mechanical and Physical Constraints}
  \small
  \begin{columns}[T]
    \column{0.58\textwidth}
    \begin{itemize}\setlength\itemsep{3pt}
      \item The optimised structure must be mechanically feasible,
            not only numerically optimal.
      \item \textbf{Static strength}: avoid yielding or fracture:
        \[
          \sigma_{\mathrm{vm}} \le \sigma_y
        \]
      \item \textbf{Fatigue constraint}: evaluate cyclic loading using
            a Goodman-type criterion \cite{zhao2023topology}:
        \[
          \frac{\sigma_a}{\sigma_e}
          +
          \frac{\sigma_m}{\sigma_{ut}}
          \le 1
        \]
      \item Mean and alternating stresses are:
        \[
          \sigma_a=\frac{\sigma_{\max}-\sigma_{\min}}{2},
          \qquad
          \sigma_m=\frac{\sigma_{\max}+\sigma_{\min}}{2}
        \]
      \item \textbf{Manufacturing}: density filtering controls minimum
            feature size and improves 3D printability.
    \end{itemize}

    \column{0.40\textwidth}
    \centering

\begin{frame}{Energy-Based Objective Function}
  \small
  \begin{columns}[T]
    \column{0.56\textwidth}
    \begin{itemize}\setlength\itemsep{3pt}
      \item The reference paper mainly maximises output displacement
            for compliant mechanisms \cite{zhao2023topology}.
      \item Our project modifies the objective toward energy storage.
      \item Stored elastic energy can be evaluated by the
            force--displacement curve:
        \[
          E_{\text{stored}}
          =
          \int_0^{\delta_{\max}} F(\delta)\,\dd{\delta}
        \]
      \item For an approximately linear elastic response:
        \[
          E_{\text{stored}}
          \approx
          \frac{1}{2}k\delta_{\max}^2
        \]
      \item Energy release efficiency:
        \[
          \eta=\frac{E_{\text{out}}}{E_{\text{in}}}
        \]
    \end{itemize}

    \column{0.42\textwidth}
    \centering
    \begin{tikzpicture}[scale=0.75, every node/.style={font=\tiny}]
      \draw[->, thick] (0,0) -- (4.2,0) node[right] {$\delta$};
      \draw[->, thick] (0,0) -- (0,3.2) node[above] {$F$};

      \draw[blue, very thick] 
        plot[smooth] coordinates {(0,0) (0.8,0.6) (1.6,1.3) (2.4,2.0) (3.4,2.8)};
      \fill[blue!20, opacity=0.7]
        (0,0) -- plot[smooth] coordinates {(0.8,0.6) (1.6,1.3) (2.4,2.0) (3.4,2.8)}

      \end{scope}
    \end{tikzpicture}
  \end{columns}
\end{frame}

\begin{frame}{Apparatus \& Materials}
  \small
  \begin{itemize}\setlength\itemsep{3pt}
    \item \textbf{Platform:} Hand wheel + lead screw compression rig (custom-built)
    \item \textbf{Force measurement:} Digital gauge, 0--50\,N, 0.1\,N resolution
    \item \textbf{Displacement:} Digital calliper, $\pm 0.01$\,mm
    \item \textbf{Material:} PLA filament (1.75\,mm), FDM 3D printing
    \item \textbf{Software:} GNU Octave, FreeCAD, Python
    \item \textbf{Environment:} Room temperature ($\sim\!25\,{}^{\circ}\mathrm{C}$)
  \end{itemize}
\end{frame}
