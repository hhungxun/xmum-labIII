\begin{frame}{Analysis Plan}
  \small
  \begin{enumerate}
    \item \textbf{Data acquisition:}
          Record force $F$ and displacement $\delta$ at 0.1\,mm
          increments during quasi-static compression.
          Repeat each topology $\ge 3$ times.
    \item \textbf{Preprocessing:}
          Zero-offset correction; apply force sensor calibration
          curve; discard first cycle (pre-conditioning).
    \item \textbf{Feature extraction:}
          \begin{itemize}
            \item Stiffness: slope of linear region of $F$--$\delta$ curve.
            \item $E_{\text{in}}$: area under loading curve (trapezoidal integration).
            \item $E_{\text{out}}$: area under unloading curve.
            \item $\eta = E_{\text{out}}/E_{\text{in}}$.
          \end{itemize}
    \item \textbf{Statistical treatment:}
          Mean $\pm$ standard deviation across trials;
          compare topologies using bar charts with error bars.
    \item \textbf{Validation:}
          Compare experimental stiffness with FEA prediction
          (if time permits, COMSOL cross-check).
  \end{enumerate}
\end{frame}
