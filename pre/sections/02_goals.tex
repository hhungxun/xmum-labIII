\begin{frame}{Project Goals}
  \begin{enumerate}
    \item \textbf{Define objective function} for elastic energy storage
          performance within a fixed planar footprint.
    \item \textbf{Generate candidate structures} via:
          \begin{itemize}
            \item Manual baseline designs (straight, folded, serpentine, curved beams)
            \item Computational topology optimisation (SIMP method)
          \end{itemize}
    \item \textbf{Fabricate and test} all candidates using 3D-printed PLA samples.
    \item \textbf{Compare} force--displacement behaviour, energy storage
          $E_{\text{in}}$, energy release $E_{\text{out}}$, and efficiency
          $\eta$ across topologies.
    \item \textbf{Analyse} how topology affects load paths, deformation modes,
          and stiffness distribution.
  \end{enumerate}
\end{frame}

\begin{frame}{Hypotheses}
  \begin{itemize}
    \item \textbf{H1:} 
    Topologies with distributed deformation (e.g. serpentine or curved structures) will achieve higher energy storage efficiency than straight beam structures.
      \begin{itemize}
        \item \textit{Rationale:} More uniform deformation reduces stress concentration and energy loss.
      \end{itemize}

    \item \textbf{H2:} 
    Topology-optimized structures will outperform manually designed baselines in terms of energy efficiency and performance stability.
      \begin{itemize}
        \item \textit{Rationale:} Optimization explores non-intuitive designs that better balance stiffness and compliance.
      \end{itemize}
  \end{itemize}
\end{frame}
