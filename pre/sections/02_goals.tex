\begin{frame}{Project Goals}
  \begin{itemize}
    \item \textbf{Goal 1:} 
Define a physically meaningful objective function for energy storage and output performance.

    \item \textbf{Goal 2:} Measure force displacement behavior to evaluate energy input, output, and efficiency.

    \item \textbf{Goal 3:} Analyze how topology affects load paths, deformation modes, and stiffness.
  \end{itemize}
\end{frame}
 
\begin{frame}{Hypotheses}
  \begin{itemize}
    \item \textbf{H1:} Topologies with distributed deformation (e.g. serpentine or curved structures) will achieve higher energy storage efficiency than straight beam structures.
      \begin{itemize}
        \item \textit{Rationale:} More uniform deformation reduces stress concentration and energy loss.
      \end{itemize}
    \item \textbf{H2:}  Topology-optimized structures will outperform manually designed baselines in terms of energy efficiency and performance stability.
      \begin{itemize}
        \item \textit{Rationale:} Optimization explores non-intuitive designs that better balance stiffness and compliance.
      \end{itemize}
  \end{itemize}
\end{frame}
