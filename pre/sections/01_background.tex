% ------------------------------------------------------------
%  Frame 1 — Compliant vs Traditional Mechanisms
% ------------------------------------------------------------
\begin{frame}{Compliant vs.\ Traditional Mechanisms}
  \small
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Traditional (rigid-body)}
    \begin{itemize}\setlength\itemsep{2pt}
      \item Rigid links + discrete joints (pin, slider, bearing)
      \item Motion from assembled kinematic pairs
      \item Friction, backlash, wear; needs lubrication
      \item Multi-part assembly; energy stored externally
            (coil spring, torsion bar)
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Compliant (flexure-based)}
    \begin{itemize}\setlength\itemsep{2pt}
      \item Single monolithic part; motion = elastic deformation
            \cite{howell2001}
      \item No sliding contacts $\Rightarrow$ no friction or wear
      \item Backlash-free, miniaturisable to MEMS
            \cite{zhu2020compliant}
      \item Energy stored \emph{in} the structure itself
    \end{itemize}
  \end{columns}

  \vspace{0.5em}
  \centering
  \begin{tikzpicture}[scale=0.55, every node/.style={font=\tiny}]
    % --- Traditional: 4-bar linkage ---
    \begin{scope}
      \draw[thick] (0,0) -- (0.6,1.4) -- (2.4,1.4) -- (3.0,0);
      \foreach \p in {(0,0),(3.0,0),(0.6,1.4),(2.4,1.4)}
        \draw[fill=white,thick] \p circle (0.10);
      \draw[thick] (-0.3,0) -- (3.3,0);
      \foreach \x in {-0.2,0.0,...,3.2}
        \draw[thin] (\x,0) -- (\x-0.15,-0.20);
      \node[below=6pt] at (1.5,-0.3) {Pin-jointed linkage};
    \end{scope}
    % --- Compliant: cross-axis flexure pivot ---
    \begin{scope}[xshift=5cm]
      \fill[blue!18] (0.0,0.05) rectangle (0.40,1.40);
      \fill[blue!18] (2.60,0.05) rectangle (3.00,1.40);
      \draw[very thick, blue] (0.40,0.25) -- (2.60,1.20);
      \draw[very thick, blue] (0.40,1.20) -- (2.60,0.25);
      \draw[thick] (-0.3,0) -- (3.3,0);
      \foreach \x in {-0.2,0.0,...,3.2}
        \draw[thin] (\x,0) -- (\x-0.15,-0.20);
      \node[below=6pt] at (1.5,-0.3) {Compliant flexure pivot};
    \end{scope}
  \end{tikzpicture}
\end{frame}

\note{%
  \begin{itemize}
    \item Frame the contrast first: traditional machine design = rigid
          parts + joints; compliant design = one elastic body.
    \item Traditional: friction at every joint, backlash from
          tolerances, parts wear; energy stored in a separate spring.
    \item Compliant: motion comes from bending the material itself,
          so no sliding contact, no lubrication, no wear.
    \item Practical wins: single-piece $\Rightarrow$ cheap to 3D-print;
          no clearance $\Rightarrow$ MEMS-scale possible; energy is
          stored \emph{inside} the structure rather than in an
          attached spring.
    \item Cost: design space is harder --- stiffness, range of motion,
          stress concentration all coupled. That coupling is exactly
          what topology choice (this project) controls.
  \end{itemize}
}


% ------------------------------------------------------------
%  Frame 2 — Planar Elastic Energy Storage
% ------------------------------------------------------------
\begin{frame}{Background: Planar Elastic Energy Storage}
  \small
  \begin{columns}[T]
    \column{0.56\textwidth}
    \begin{itemize}\setlength\itemsep{3pt}
      \item Compliant flexures store and release energy via
            reversible elastic deformation \cite{howell2001}.
      \item Applications: MEMS actuators, soft grippers, watch
            springs, miniature energy harvesters
            \cite{zhu2020compliant}.
      \item Often confined to thin planar footprints
            (PCB-mounted, wearable, additively manufactured)
            \cite{tong2023spring}.
      \item Key metric --- \textbf{energy storage efficiency}:
            \[
              \eta = \frac{E_{\text{out}}}{E_{\text{in}}}
                   = \frac{\int F_{\text{out}}\,\dd{x}}
                          {\int F_{\text{in}}\,\dd{x}}
            \]
    \end{itemize}

    \column{0.42\textwidth}
    \centering

    %  -- shows planarity, footprint constraint, and elastic deflection

\note{%
  \begin{itemize}
    \item Zoom in on energy storage. Flexure deforms reversibly;
          stored elastic energy released on unload.
    \item Application list emphasises why \emph{planar} matters:
          PCBs, wearables, FDM-printed parts are locked to a shallow
          box --- same constraint we use later
          (100$\times$100$\times$3 mm).
    \item \textbf{Top figure:} planar arch (``bow'') spring inside a
          dashed footprint box. Central load $F$ pushes the arch down
          by $\delta$. Picks up planarity + footprint constraint
          + elastic deflection in one image.
    \item \textbf{Bottom figure:} $F(\delta)$ loading curve. Shaded
          area $= E_{\text{in}} = \int F\,\dd\delta$, the integral on
          the slide.
    \item $E_{\text{out}}$ is the analogous area under the
          \emph{unloading} branch (omitted here for clarity); the
          gap between the two = hysteresis loss. Steady-state $\eta$
          reported (first cycle discarded for PLA Mullins effect).
    \item Same quantity is called \emph{round-trip efficiency} in
          battery / flywheel literature; \emph{loss factor}
          ($\tan\delta$) in viscoelasticity. Same physics.
  \end{itemize}
}

%  Frame 3 — Motivation & Research Gap
% ------------------------------------------------------------
\begin{frame}{Motivation \& Research Gap}
  \begin{itemize}\setlength\itemsep{3pt}
    \item Within a fixed planar footprint, geometry dictates the load
          path and strain distribution, hence the stored elastic
          energy at a given material volume \cite{howell2001}.
    \item Folded / serpentine layouts lengthen the effective beam
          inside the same area, distributing strain more uniformly
          and delaying yield at the maximum-stress fibre.
    \item Topology optimisation (SIMP) generates non-intuitive
          material layouts that minimise compliance under a volume
          constraint
          \cite{bendsoe2003topology, sigmund2001, andreassen2011}.
    \item \textbf{Gap:} experimental comparisons of fold count and
          pattern family at \emph{equal volume} and \emph{equal
          footprint} are scarce; most topology-optimised planar
          springs are reported only numerically
          \cite{tong2023spring, zhu2020compliant}.
  \end{itemize}
\end{frame}

\note{%
  \begin{itemize}
    \item Punchline: at fixed footprint and fixed material, geometry
          alone moves $\eta$ --- because it changes where the strain
          concentrates.
    \item Folded / serpentine = more beam length stuffed in the same
          box $\Rightarrow$ strain spread along longer fibre
          $\Rightarrow$ higher elastic capacity before yield.
    \item SIMP (Bendsøe--Sigmund) lets the algorithm decide where
          material lives. Methodology slide will give the math; here
          we just flag it as the second design path.
    \item Gap framing: most planar-spring topology-optimisation
          papers stop at simulation; few compare against hand-designed
          baselines under matched volume + footprint. This is the
          niche the project occupies.
  \end{itemize}
}
